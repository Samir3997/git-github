import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertNull;

class DateUtilTest {

    @Test
    public void testGetDateTimeFromStringWithFormat_ValidFormat() {
        String dateString = "2023-12-31T23:59:59";
        String format = "yyyy-MM-dd'T'HH:mm:ss";
        LocalDateTime expectedDateTime = LocalDateTime.parse(dateString, DateTimeFormatter.ofPattern(format));
        LocalDateTime actualDateTime = DateUtil.getDateTimeFromStringWithFormat(dateString, format);
        assertEquals(expectedDateTime, actualDateTime);
    }

    @Test
    public void testGetDateTimeFromStringWithFormat_InvalidFormat() {
        String dateString = "invalid date string";
        String format = "yyyy-MM-dd'T'HH:mm:ss";
        LocalDateTime actualDateTime = DateUtil.getDateTimeFromStringWithFormat(dateString, format);
        assertNull(actualDateTime);
    }

}
