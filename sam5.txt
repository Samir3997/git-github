import org.junit.jupiter.api.Test;

import java.time.ZoneId;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertNull;

class DateUtilTest {

    @Test
    public void testGetCurrentDateTimeByZone_ValidZone() {
        String zone = "Asia/Kolkata";
        LocalDateTime expectedDateTime = LocalDateTime.now(ZoneId.of(zone));
        LocalDateTime actualDateTime = DateUtil.getCurrentDateTimeByZone(zone);
        assertEquals(expectedDateTime, actualDateTime);
    }

    @Test
    public void testGetCurrentDateTimeByZone_InvalidZone() {
        String zone = "invalid zone";
        LocalDateTime actualDateTime = DateUtil.getCurrentDateTimeByZone(zone);
        assertNull(actualDateTime);
    }

}
