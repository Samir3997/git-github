import static org.junit.jupiter.api.Assertions.assertEquals;

import java.time.LocalDate;

import org.junit.jupiter.api.Test;

public class DateUtilTest {

    @Test
    public void testGetSkeyFromLocalDate() {
        // Test case 1: A valid date
        LocalDate localDate1 = LocalDate.of(2023, 11, 10);
        long expectedSkey1 = 20231110L;
        long actualSkey1 = DateUtil.getSkeyFromLocalDate(localDate1);
        assertEquals(expectedSkey1, actualSkey1);

        // Test case 2: A date with leading zeros
        LocalDate localDate2 = LocalDate.of(2001, 02, 03);
        long expectedSkey2 = 20010203L;
        long actualSkey2 = DateUtil.getSkeyFromLocalDate(localDate2);
        assertEquals(expectedSkey2, actualSkey2);

        // Test case 3: A date with a year greater than 9999
        LocalDate localDate3 = LocalDate.of(9999, 12, 31);
        long expectedSkey3 = 99991231L;
        long actualSkey3 = DateUtil.getSkeyFromLocalDate(localDate3);
        assertEquals(expectedSkey3, actualSkey3);

        // Test case 4: A null date
        LocalDate localDate4 = null;
        long expectedSkey4 = 0L;
        long actualSkey4 = DateUtil.getSkeyFromLocalDate(localDate4);
        assertEquals(expectedSkey4, actualSkey4);
    }
}
