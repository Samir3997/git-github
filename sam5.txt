@Test
public void testAsJsonString() throws JsonProcessingException {
    ObjectMapper mockMapper = Mockito.mock(ObjectMapper.class);
    String mockJson = "{\"<key>\":\"<value>\"}";
    Object mockObject = new Object();
    Mockito.when(mockMapper.writeValueAsString(mockObject)).thenReturn(mockJson);
    ConversionUtil.asJsonString(mockObject);
    Mockito.verify(mockMapper).writeValueAsString(mockObject);
}
