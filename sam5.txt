@Test
public void testGetDateFromString_ValidFormat() {
    String dateString = "23092024";
    LocalDate expectedDate = LocalDate.parse(dateString, DateTimeFormatter.ofPattern("yyMMdd"));
    LocalDate actualDate = DateUtil.getDateFromString(dateString);
    assertEquals(expectedDate, actualDate);
}

@Test
public void testGetDateFromString_InvalidFormat() {
    String dateString = "invalid date string";
    LocalDate actualDate = DateUtil.getDateFromString(dateString);
    assertNull(actualDate);
}
