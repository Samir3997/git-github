import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertNull;

class DateUtilTest {

    @Test
    public void testGetDateFromStringYYYYMMDD_ValidFormat() {
        String dateString = "20231231";
        LocalDate expectedDate = LocalDate.parse(dateString, DateTimeFormatter.ofPattern("yyyyMMdd"));
        LocalDate actualDate = DateUtil.getDateFromStringYYYYMMDD(dateString);
        assertEquals(expectedDate, actualDate);
    }

    @Test
    public void testGetDateFromStringYYYYMMDD_InvalidFormat() {
        String dateString = "invalid date string";
        LocalDate actualDate = DateUtil.getDateFromStringYYYYMMDD(dateString);
        assertNull(actualDate);
    }

    // Add more test cases with different valid and invalid date strings
}
