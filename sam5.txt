@Test
public void blockMT103MsgITypeDebitMsgTest() {
    // Test when BIC is exactly "MBEAMYKL"
    FinMT103 finMT103 = new FinMT103();
    Block4Tag50 tag50 = new Block4Tag50();
    tag50.setBlock4Tag50aBic("MBEAMYKL123");
    finMT103.setTags50(tag50);

    Boolean result = mt103Handler.blockMT103MsgITypeDebitMsg(finMT103, "MYR");
    assertTrue(result, "Message should be blocked for BIC 'MBEAMYKL123'");

    // Test when BIC is exactly "MBEAMY21"
    tag50.setBlock4Tag50aBic("MBEAMY21ABC");
    result = mt103Handler.blockMT103MsgITypeDebitMsg(finMT103, "MYR");
    assertTrue(result, "Message should be blocked for BIC 'MBEAMY21ABC'");

    // Test when BIC is not blocked
    tag50.setBlock4Tag50aBic("OTHERBIC01");
    result = mt103Handler.blockMT103MsgITypeDebitMsg(finMT103, "MYR");
    assertFalse(result, "Message should not be blocked for BIC 'OTHERBIC01'");
}
