import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.assertEquals;

class DateUtilTest {

    @Test
    public void testConvertLdatetoLdatetime_ValidInput() {
        LocalDate localDate = LocalDate.of(2023, 12, 31);
        String hourmin = "2359";
        LocalDateTime expectedDateTime = LocalDateTime.of(2023, 12, 31, 23, 59);
        LocalDateTime actualDateTime = DateUtil.convertLdatetoLdatetime(localDate, hourmin);
        assertEquals(expectedDateTime, actualDateTime);
    }

    @Test
    public void testConvertLdatetoLdatetime_InvalidHourminFormat() {
        LocalDate localDate = LocalDate.of(2023, 12, 31);
        String hourmin = "invalid hourmin";
        LocalDateTime actualDateTime = DateUtil.convertLdatetoLdatetime(localDate, hourmin);
        assertNull(actualDateTime);
    }

}
