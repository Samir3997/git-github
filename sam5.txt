import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class ThrowMsgBlockedExceptionTest {

    @Test
    void testThrowMsgBlockedException() {
        // Create an instance of the class under test (if applicable)
        ThrowMsgBlockedException classUnderTest = new ThrowMsgBlockedException(); // Optional

        // Call the function and capture the thrown exception
        ILMException thrownException = assertThrows(ILMException.class, () -> {
            classUnderTest.throwMsgBlockedException();
        });

        // Verify the exception's properties
        assertEquals(ExceptionConstants.FAILED, thrownException.getStatus());
        assertEquals(ExceptionConstants.MSGBLOCKED, thrownException.getMessage());

    }
}
