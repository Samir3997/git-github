        MTField field2 = new MTField();
        field2.setFieldName("fieldName2");

        field2.setFieldType("type2");

        // Create MTSequence instances
        MTSequence sequence1 = new MTSequence();
        sequence1.setSeqName("seqName1");
        sequence1.setMtFields(Arrays.asList(field1, field2));

        MTSequence sequence2 = new MTSequence();
        sequence2.setSeqName("seqName2");
        sequence2.setMtFields(Collections.singletonList(field2));

        List<MTSequence> sequences = Arrays.asList(sequence1, sequence2);

        // Define input parameters
        String seqName = "seqName1";
