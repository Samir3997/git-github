import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

class YourClassNameTest {

    @Test
    void testGetLocalDateTimeFromString() {
        // Test valid input
        String validDate = "Jan 25 2024 12:30 PM";
        LocalDateTime expectedDateTime = LocalDateTime.of(2024, 1, 25, 12, 30);
        LocalDateTime actualDateTime = YourClassName.getLocalDateTimeFromString(validDate);
        assertEquals(expectedDateTime, actualDateTime);

        // Test invalid input (incorrect format)
        String invalidDate1 = "2024-01-25 12:30";
        assertThrows(DateTimeParseException.class, () -> YourClassName.getLocalDateTimeFromString(invalidDate1));

        // Test invalid input (invalid date)
        String invalidDate2 = "February 30 2024 12:30 PM";
        assertThrows(DateTimeParseException.class, () -> YourClassName.getLocalDateTimeFromString(invalidDate2));
    }
}
