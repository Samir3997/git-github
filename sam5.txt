import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertNotNull;
import static org.junit.jupiter.api.Assertions.assertTrue;

import org.junit.jupiter.api.Test;

import com.maybank.ilm.core.util.MTFieldDataTypeMap;

public class MTFieldDataTypeMapTest {

    private final MTFieldDataTypeMap map = new MTFieldDataTypeMap();

    @Test
    public void testFieldDTMap() {
        // Test a few sample mappings
        assertEquals("ILMCoreConstants.STRING", map.getFieldDTMap().get("16R_BLOCKNAME"));
        assertEquals("ILMCoreConstants.BIGDECIMAL", map.getFieldDTMap().get("98E_DECIMALS"));
        assertEquals("ILMCoreConstants.STRING", map.getFieldDTMap().get("70E_NARRATIVE_10"));
    }

    @Test
    public void testParentChildSeqMap() {
        // Test parent-child relationships
        assertTrue(map.getParentChildSeqMap().get("A").contains("A1_LINK_R"));
        assertTrue(map.getParentChildSeqMap().get("E").contains("E1_SETPRTY_R"));
    }

    @Test
    public void testSeqTagMap() {
        // Test sequence tags
        assertTrue(map.getSeqTagMap().get("A").contains("16R"));
        assertTrue(map.getSeqTagMap().get("B1").contains("12B"));
    }
