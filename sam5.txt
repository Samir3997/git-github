import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.MockitoAnnotations;

import java.lang.reflect.Method;
import java.math.BigDecimal;
import java.util.Arrays;
import java.util.List;

public class RentasBalancingDaoImplTest {

    @InjectMocks
    private RentasBalancingDaoImpl rentasBalancingDaoImpl;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    public void testConvertToList() throws Exception {
        // Given: A sample input list of Object arrays
        List<Object[]> inputList = Arrays.asList(
            new Object[]{new BigDecimal("100.50"), new BigDecimal("200.75")},
            new Object[]{new BigDecimal("300.25"), new BigDecimal("400.90")}
        );

        // Accessing the private method via reflection
        Method method = RentasBalancingDaoImpl.class.getDeclaredMethod("convertToList", List.class);
        method.setAccessible(true);

        // When: Invoking the private convertToList method
        @SuppressWarnings("unchecked")
        List<BigDecimal> result = (List<BigDecimal>) method.invoke(rentasBalancingDaoImpl, inputList);

        // Then: Validate the output
        assertNotNull(result);
        assertEquals(4, result.size());
        assertEquals(new BigDecimal("100.50"), result.get(0));
        assertEquals(new BigDecimal("200.75"), result.get(1));
        assertEquals(new BigDecimal("300.25"), result.get(2));
        assertEquals(new BigDecimal("400.90"), result.get(3));
    }
}
