package com.maybank.ilm.entity;

import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.SequenceGenerator;
import javax.persistence.Table;

@Entity
@Table(name = "PRC_MT_FIELD")
public class MTField {

	@Id
	@Column(name = "N_MT_FIELD_SKEY")
	@SequenceGenerator(name = "mtFieldSkey", sequenceName = "PRC_MT_FIELD_SEQ", allocationSize = 1)
	@GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "mtFieldSkey")
	private Long mtFieldSkey;
	
	@Column(name = "N_MT_SEQUENCE_SKEY")
	private Long mtSeqSkey;
	
	@Column(name = "V_FIELD_NAME")
	private String fieldName;
	
	@Column(name = "V_FIELD_VALUE")
	private String fieldValue;
	
	@Column(name = "V_FIELD_TYPE")
	private String fieldType;

	public Long getMtFieldSkey() {
		return mtFieldSkey;
	}

	public void setMtFieldSkey(Long mtFieldSkey) {
		this.mtFieldSkey = mtFieldSkey;
	}

	public Long getMtSeqSkey() {
		return mtSeqSkey;
	}

	public void setMtSeqSkey(Long mtSeqSkey) {
		this.mtSeqSkey = mtSeqSkey;
	}

	public String getFieldName() {
		return fieldName;
	}

	public void setFieldName(String fieldName) {
		this.fieldName = fieldName;
	}

	public String getFieldValue() {
		return fieldValue;
	}

	public void setFieldValue(String fieldValue) {
		this.fieldValue = fieldValue;
	}

	public String getFieldType() {
		return fieldType;
	}

	public void setFieldType(String fieldType) {
		this.fieldType = fieldType;
	}

	@Override
	public String toString() {
		return "MTField [mtFieldSkey=" + mtFieldSkey + ", mtSeqSkey=" + mtSeqSkey + ", fieldName=" + fieldName
				+ ", fieldValue=" + fieldValue + ", fieldType=" + fieldType + "]";
	}
}
