package com.maybank.ilm.entity;

import java.util.List;

import javax.persistence.CascadeType;
import javax.persistence.Column;
import javax.persistence.Entity;
import javax.persistence.FetchType;
import javax.persistence.GeneratedValue;
import javax.persistence.GenerationType;
import javax.persistence.Id;
import javax.persistence.JoinColumn;
import javax.persistence.ManyToOne;
import javax.persistence.OneToMany;
import javax.persistence.SequenceGenerator;
import javax.persistence.Table;

import org.hibernate.annotations.LazyCollection;
import org.hibernate.annotations.LazyCollectionOption;

@Entity
@Table(name = "PRC_MT_SEQUENCE")
public class MTSequence {

	@Id
	@Column(name = "N_MT_SEQUENCE_SKEY")
	@SequenceGenerator(name = "mtSeqSkey", sequenceName = "PRC_MT_SEQUENCE_SEQ", allocationSize = 1)
	@GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "mtSeqSkey")
	private Long mtSeqSkey;
	
	@Column(name = "N_MT_SKEY")
	private Long mtSkey;
	
	@Column(name = "V_SEQUENCE_NAME")
	private String seqName;
	
	@ManyToOne(fetch = FetchType.LAZY, optional=true)
	@JoinColumn(name="N_PARENT_SEQ_SKEY")
	private MTSequence parentSeq;
	
	@OneToMany(fetch = FetchType.LAZY, cascade = CascadeType.ALL)
	@LazyCollection(LazyCollectionOption.FALSE)
	@JoinColumn(name = "N_MT_SEQUENCE_SKEY")
	private List<MTField> mtFields;
	
	@OneToMany(mappedBy = "parentSeq", fetch = FetchType.LAZY, cascade = CascadeType.ALL)
	@LazyCollection(LazyCollectionOption.FALSE)
	private List<MTSequence> mtSequences;

	public Long getMtSeqSkey() {
		return mtSeqSkey;
	}

	public void setMtSeqSkey(Long mtSeqSkey) {
		this.mtSeqSkey = mtSeqSkey;
	}

	public Long getMtSkey() {
		return mtSkey;
	}

	public void setMtSkey(Long mtSkey) {
		this.mtSkey = mtSkey;
	}

	public String getSeqName() {
		return seqName;
	}

	public void setSeqName(String seqName) {
		this.seqName = seqName;
	}

	public List<MTField> getMtFields() {
		return mtFields;
	}

	public void setMtFields(List<MTField> mtFields) {
		this.mtFields = mtFields;
	}

	public List<MTSequence> getMtSequences() {
		return mtSequences;
	}

	public void setMtSequences(List<MTSequence> mtSequences) {
		this.mtSequences = mtSequences;
	}

	public MTSequence getParentSeq() {
		return parentSeq;
	}

	public void setParentSeq(MTSequence parentSeq) {
		this.parentSeq = parentSeq;
	}

	@Override
	public String toString() {
		return "MTSequence [mtSeqSkey=" + mtSeqSkey + ", mtSkey=" + mtSkey + ", seqName=" + seqName + ", parentSeq="
				+ parentSeq + ", mtFields=" + mtFields + ", mtSequences=" + mtSequences + "]";
	}	
}
